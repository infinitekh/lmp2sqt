132 void AccumSpacetimeCorr ( int nCol)
133 {
134   real fac;
135   int j,  nb, nr, n;
136   for (nb = 0; nb < nBuffCorr; nb ++) {
137     if (tBuf[nb].count == nValCorr) {
138       // S(q,t), M(q,t) part
139       for (j = 0; j < 3 * nFunCorr; j ++) {
140         for (n = 0; n < nValCorr; n ++)
141           avAcfST[j][n] += tBuf[nb].acfST[j][n];
142       }
143       // Diffuse Part
144       for (j = 0; j < nValCorr; j ++) {
145         rrDiffuseAv[j] += tBuf[nb].rrDiffuse[j];
146         for ( nr=0; nr<nFunCorr; nr++)
147           avDrTable[nr][j] += tBuf[nb].DrTable[nr][j];
148       }
149       // buffer nb reset
150       tBuf[nb].count = 0;
151       ++ countCorrAv;
152       if (countCorrAv == limitCorrAv) {
153         for (j = 0; j < 3 * nFunCorr; j ++) {
154           for (n = 0; n < nValCorr; n ++)
155             avAcfST[j][n] /= 3. * nCol * limitCorrAv;
156         }
157 
158         fac = 1./ ( DIM * 2 * nCol * deltaT * limitCorrAv);
159         for (j = 1; j < nValCorr; j ++) {
160           rrDiffuseAv[j] *= fac/ j;
161           for ( nr=0; nr<nFunCorr; nr++)
162             avDrTable[nr][j] *= factorDr[nr];
163         }
164         PrintSpacetimeCorr (stdout);
165         ZeroSpacetimeCorr ();
166       }
167     }
168   }
169 }

172 void InitSpacetimeCorr ()
173 {
174   int nb;
175   if (nBuffCorr > nValCorr) {
176     fputs("Error nBuffCorr> nValCorr", stderr);
177     exit(1);
178   }
179 
180   for (nb = 0; nb < nBuffCorr; nb ++){
181     tBuf[nb].count = - nb * nValCorr / nBuffCorr;
182     tBuf[nb].countDiff = - nb * nValCorr / nBuffCorr;
183   }
184   ZeroSpacetimeCorr ();

186 void ZeroSpacetimeCorr ()
187 {
188   int j, n, nr;
189   countCorrAv = 0;
190   for (j = 0; j < 3 * nFunCorr; j ++) {
191     for (n = 0; n < nValCorr; n ++) avAcfST[j][n] = 0.;
192   }
193 
194   countDiffuseAv = 0;
195   for (j = 0; j < nValCorr; j ++) rrDiffuseAv[j] = 0.;
196   for (j = 0; j < nValCorr; j ++)
197     for (nr = 0; nr < nFunCorr; nr ++) avDrTable[nr][j]= 0.;

199 void EvalOtherInformation ()
200 {   // this evaluation yield analysis.c
201 #define Fqt_FIX_q avAcfST[3*(j) +2] 
202   int j,  n,  ppT, pT, cT, nT, nnT;
203   extern real kVal;
204   real kVal2 = kVal*kVal;
205   n=0; nnT = n+2; nT = n+1; cT = n; {  //Forward O(h^2)
206     for (j = 0; j < nFunCorr; j ++) {
207       valGammaQT[j][n]=    (-(Fqt_FIX_q[nnT]) +4.*(Fqt_FIX_q[nT]) -3.*(Fqt_FIX_q[cT]) )/ (2.0* deltaT*Fqt_FIX_q[cT]);
208       valDqt [j][n] = - valGammaQT[j][n] / (kVal2*j*j) ;
209     }
210   }
211   for (n = 1; n < nValCorr-1; n ++) {     /* centerd O(h^2) */
212     pT = n-1; nT = n+1; cT = n;
213     for (j = 0; j < nFunCorr; j ++) {
214       valGammaQT[j][n] = ( (Fqt_FIX_q[nT]) -(Fqt_FIX_q[pT]) )/ (2.0* deltaT*Fqt_FIX_q[cT]);
215       valDqt [j][n] = - valGammaQT[j][n] / (kVal2*j*j) ;
216     }
217   }
218   n= nValCorr-1; ppT = n-2; pT = n-1; cT = n; { /* Backward O(h^2) */
219     for (j = 0; j < nFunCorr; j ++) {
220       valGammaQT[j][n] = (+(3.*Fqt_FIX_q[cT]) -4.*(Fqt_FIX_q[pT]) +(Fqt_FIX_q[ppT]) )/ (2.0* deltaT*Fqt_FIX_q[cT]);
221       valDqt [j][n] = - valGammaQT[j][n] / (kVal2*j*j) ;
222     }
223   }
224 
225 }

350 void EvalSpacetimeCorr(Snapshot* snap)
351 {
352   real b, c, c0, c1, c2, s, s1, s2, w;
353   extern real kVal;
354   int j, k, m, n, nb, nc, ni, nv, nr;
355   real r[3], mu[3]; // L[3];
356 
357   VecR3 dr;
358   real deltaR;
359   int i_Dr;
360   real L = snap->box.xhigh- snap->box.xlow;
361   g_Vol  = L*L*L;
362   int nCol = snap->n_atoms;
363   static int first_run = 0;
364   if (first_run ==0 )
365     Alloc_more(nCol);
366   first_run = 1;
367   atom* col_i;
368   // zero initalize current time value
369   for (j = 0; j < 24 * nFunCorr; j ++) valST[j] = 0.;
370 // we assume L0=L1 = L2 
371 /*  L[0] = snap->box.xhigh- snap->box.xlow;
372  *  L[1] = snap->box.yhigh- snap->box.ylow;
373  *  L[2] = snap->box.zhigh- snap->box.zlow;
374  */
375   kVal = 2.*M_PI / L;
376 
377 //  kVal = 2. * M_PI / nFunCorr;
378 //  Begin Calculatie current time information
379   /*-----------------------------------------------------------------------------
380    *  Direct calculate  rho(q)
381    *-----------------------------------------------------------------------------*/
382   for (n=0; n<nCol; n++) {
383     col_i = &(snap->atoms[n]);
384 /*    r[0] = col_i->x; r[0] = r[0] - L* floor(r[0]/L)- L/2.;  
385  *    r[1] = col_i->y; r[1] = r[1] - L* floor(r[1]/L)- L/2.;  
386  *    r[2] = col_i->z; r[2] = r[2] - L* floor(r[2]/L)- L/2.;  
387  */
388     r[0] = col_i->x;
389     r[1] = col_i->y;
390     r[2] = col_i->z;
391     mu[0] = col_i->mux;
392     mu[1] = col_i->muy;
393     mu[2] = col_i->muz;
394     j = 0;
395     // 
396     for (k = 0; k < 3; k ++) {
397       for (m = 0; m < nFunCorr; m ++) {
398         if (m == 0) {
399           b = kVal * r[k];
400           c = cos (b);
401           s = sin (b);
402           c0 = c;
403         } else if (m == 1) {
404           c1 = c;
405           s1 = s;
406           c = 2. * c0 * c1 - 1.;
407           s = 2. * c0 * s1;
408         } else {
409           c2 = c1;
410           s2 = s1;
411           c1 = c;
412           s1 = s;
413           c = 2. * c0 * c1 - c2;
414           s = 2. * c0 * s1 - s2;
415         }
416         valST[j ++] += mu[0] * c;
417         valST[j ++] += mu[0] * s;
418         valST[j ++] += mu[1] * c;
419         valST[j ++] += mu[1] * s;
420         valST[j ++] += mu[2] * c;
421         valST[j ++] += mu[2] * s;
422         valST[j ++] += c;
423         valST[j ++] += s;
424       }
425     }
426   }
427   // End Calculate Current time value
428   // Begin Two time corrlation function
429   for (nb = 0; nb < nBuffCorr; nb ++) {
430     if (tBuf[nb].count == 0) {
431       /*-----------------------------------------------------------------------------
432        *   t_w information 
433        *-----------------------------------------------------------------------------*/
434       for (n=0; n<nCol; n++) {
435         col_i = &(snap->atoms[n]);
436         tBuf[nb].orgR[n].x = col_i->x;
437         tBuf[nb].orgR[n].y = col_i->y;
438         tBuf[nb].orgR[n].z = col_i->z;
439         tBuf[nb].rTrue[n].x = col_i->x;
440         tBuf[nb].rTrue[n].y = col_i->y;
441         tBuf[nb].rTrue[n].z = col_i->z;
442       }
443       for (j = 0; j < 24 * nFunCorr; j ++)
444         tBuf[nb].orgST[j] = valST[j];
445     } // End   buffer count ==0
446 
447     if (tBuf[nb].count >= 0) {
448       /*-----------------------------------------------------------------------------
449        *  Get Delta_r(t)
450        *-----------------------------------------------------------------------------*/
451       ni = tBuf[nb].count;
452       /*-----------------------------------------------------------------------------
453        *  Zero initializing
454        *-----------------------------------------------------------------------------*/
455       tBuf[nb].rrDiffuse[ni]= 0.;
456       for ( nr=0; nr<nFunCorr; nr++)
457         tBuf[nb].DrTable[nr][ni] =0;
458 
459       /*-----------------------------------------------------------------------------
460        *  Calculation at this time
461        *-----------------------------------------------------------------------------*/
462 
463       for (n=0; n<nCol; n++) {
464         col_i = &(snap->atoms[n]);
465         dr.x =  col_i->x-tBuf[nb].orgR[n].x ;
466         dr.y =  col_i->y-tBuf[nb].orgR[n].y ;
467         dr.z =  col_i->z-tBuf[nb].orgR[n].z ;
468         deltaR = sqrt(dr.x*dr.x+dr.y*dr.y+dr.z*dr.z);
469         //tBuf[nb].rrDiffuse[ni] += deltaR;     /* 이부분을 잘못함... */
470         tBuf[nb].rrDiffuse[ni] += deltaR*deltaR;
471 
472         i_Dr    = (int) (deltaR/rVal);
473         if (i_Dr<nFunCorr)
474           tBuf[nb].DrTable[i_Dr][ni] ++;
475       }
476 
477       /*-----------------------------------------------------------------------------
478        *  two time correlation 
479        *-----------------------------------------------------------------------------*/
480       for (j = 0; j < 3 * nFunCorr; j ++)
481         tBuf[nb].acfST[j][tBuf[nb].count] = 0.;
482 
483       for (j = 0,k = 0; k < 3; k ++) { // 3 loop
484         for (m = 0; m < nFunCorr; m ++) {
485           for (nc = 0; nc < 4; nc ++) {
486             nv = 3 * m + 2;
487             if (nc < 3) {                       /*  */
488               //              w = Sqr (kVal * (m + 1));
489               w=1.0;
490               /*-----------------------------------------------------------------------------
491                *  nv %3 = >> 0 - longitunidal, 1 - transverse, 2 - density 
492                *-----------------------------------------------------------------------------*/
493               -- nv;              // transverse  3*m +1
494               if (nc == k) -- nv; // longitudinal 3*m
495               else w = 0.5;
496               //              else w *= 0.5;
497             } else w = 1.;        // density   3*m+2
498             // cos(q*r(t)) cos(q*r(t_w) +sin sin
499             tBuf[nb].acfST[nv][tBuf[nb].count] +=
500               w * (valST[j] * tBuf[nb].orgST[j] +
501                   valST[j + 1] * tBuf[nb].orgST[j + 1]);
502             j += 2;
503           } // for nc
504         } // for m
505       }   // for k   // total 12 loop j+=2, 
506       assert( j== 24*nFunCorr);
507     }    // End buffer count >=0
508     ++ tBuf[nb].count;
509   }
510   AccumSpacetimeCorr (nCol );
511 }

513 void AllocArray ()
514 {
515   int nb;
516   AllocMem (valST, 24 * nFunCorr, real);
517   AllocMem2 (avAcfST, 3 * nFunCorr, nValCorr, real);
518   AllocMem2 (valDqt,  nFunCorr, nValCorr, real);
519   AllocMem2 (valGammaQT,  nFunCorr, nValCorr, real);
520   AllocMem (tBuf, nBuffCorr, TBuf);
521   for (nb = 0; nb < nBuffCorr; nb ++) {
522     AllocMem (tBuf[nb].orgST, 24 * nFunCorr, real);
523     AllocMem2 (tBuf[nb].acfST, 3 * nFunCorr, nValCorr, real);
524   }
525   // AllocArray for Diffuse ()
526   AllocMem (rrDiffuseAv, nValCorr, real);
527   AllocMem2 (avDrTable, nFunCorr,nValCorr, real);
528 }
529 void Alloc_more (int  nCol) {
530   int nb,nr; real rho0, shell_Vol;
531   for (nb = 0; nb < nBuffCorr; nb ++) {
532     AllocMem (tBuf[nb].orgR, nCol, VecR3);
533     AllocMem (tBuf[nb].rTrue, nCol,VecR3);
534     AllocMem (tBuf[nb].rrDiffuse, nValCorr, real);
535     AllocMem2 (tBuf[nb].DrTable, nFunCorr,nValCorr, int);
536   }
537   AllocMem (factorDr, nFunCorr, real);
538 
539   rho0 = nCol/g_Vol;
540   for (nr = 0; nr < nFunCorr; nr ++) {
541     if (nr ==0) shell_Vol = 4*M_PI /3. * pow(rVal,3);
542     else shell_Vol = 4*M_PI * pow(rVal,3)* (nr*nr + 1./12.);
543 
544     factorDr[nr] = 1./( pow(rho0,2) * g_Vol *shell_Vol*limitCorrAv);
545 /*    printf("rho0=%.2e, Vol=%.2e, shell_Vol=%.2e, factorDr=%.2e\n", 
546  *        rho0,g_Vol,shell_Vol,factorDr[nr]);
547  */
548 
549   }
550 
551 }

583   for (k = 0; k < sizeof (nameList) / sizeof (NameList); k ++)
584     nameList[k].vStatus = 0;
585   ok = 1;
586   while (1) {
587     fgets (buff, 80, fp);
588     if (feof (fp)) break;
589     token = strtok (buff, " \t\n");
590     if (! token) break;
591     match = 0;
592     for (k = 0; k < sizeof (nameList) / sizeof (NameList); k ++) {
593       if (strcmp (token, nameList[k].vName) == 0) {
594         match = 1;
595         if (nameList[k].vStatus == 0) {
596           nameList[k].vStatus = 1;
597           for (j = 0; j < nameList[k].vLen; j ++) {
598             token = strtok (NULL, ", \t\n");
599             if (token) {
600               switch (nameList[k].vType) {
601                 case N_I:
602                   *NP_I = atol (token);
603                   break;
604                 case N_R:
605                   *NP_R = atof (token);
606                   break;
607               }
608             } else {
609               nameList[k].vStatus = 2;
610               ok = 0;
611             }
612           }
613           token = strtok (NULL, ", \t\n");
614           if (token) {
615             nameList[k].vStatus = 3;
616             ok = 0;
617           }
618           break;
619         } else {
620           nameList[k].vStatus = 4;
621           ok = 0;
622         }
623       }
624     }
625     if (! match) ok = 0;
626   }
627   fclose (fp);
628 
629   if(nBuffCorr > nValCorr ) nBuffCorr = nValCorr;
630 
631   for (k = 0; k < sizeof (nameList) / sizeof (NameList); k ++) {
632     if (nameList[k].vStatus != 1) ok = 0;
633   }
634   return (ok);
635 }
